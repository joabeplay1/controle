// Função que realmente inicializa o emulador
function iniciarJogoReal(nomeArquivo, sistema) {
    // 1. Esconde a tela de lista e mostra a tela de emulação
    document.getElementById('home-screen').classList.remove('active');
    document.getElementById('game-play-screen').classList.add('active');

    // 2. Configuração do motor EJS
    const config = {
        system: sistema, 
        url: "https://joabeplay1.github.io/controle/roms/" + nomeArquivo,
        container: "emulator", // Nome da div onde o jogo vai rodar
        path: "https://cdn.emulatorjs.org/stable/data/"
    };

    // 3. Inicializa o emulador
    window.EJS_player = new EJS(config);
}

function sairJogo() {
    if (window.EJS_player) {
        window.EJS_player.destroy();
        window.EJS_player = null;
    }
    location.reload(); // Recarrega a página para limpar o emulador completamente
}
